125      /**
126       * This method receives input from the user and completes a calculation with the input
127       * @param symbol  which type of calculation to complete: either + - * /
128       * @param decimal1 the integer in the calculation
129       * @param decimal2 the second integer in the calculation
130       * @return the output of the calculation
131       */
132      public static int getTotal(char symbol, int decimal1, int decimal2) { //method to calculate the output based upon symbol input
133          int total = 0;
134          if(symbol == '+'){   //completing a calculation
135              total = decimal1 + decimal2;
136          }else if(symbol == '-'){
137              total = decimal1 - decimal2;
138          }else if(symbol == '*'){
139              total = decimal1 * decimal2;
140          }else if(symbol == '/')
141              total = decimal1 / decimal2;
142  
143          return total;
144      }
